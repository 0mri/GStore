from backend.api.product.models import Product
import math
from lorem.text import TextLorem
from random import random



for i in range(0,1000):
	lorem = TextLorem(srange=(2,5))
	length = int(len(lorem.sentence()))
	p_name = lorem.sentence()[:(length-1)]
	id = math.floor(random()*10)+1
	try: 
		Product.objects.created_by_cat_id(p_name ,id)
	except:
		pass
